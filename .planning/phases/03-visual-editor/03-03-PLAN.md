import type { ToolkitRead } from '../types'

type ArgMode = 'literal' | 'param' | 'flag'

interface Props {
  value: unknown
  toolkit: ToolkitRead | null
  onChange: (updated: unknown) => void
}

function detectMode(value: unknown): ArgMode {
  if (value !== null && typeof value === 'object') {
    if ('param' in (value as object)) return 'param'
    if ('flag' in (value as object)) return 'flag'
  }
  return 'literal'
}

export default function ArgInput({ value, toolkit, onChange }: Props) {
  const mode = detectMode(value)
  const paramKeys = Object.keys(toolkit?.params_schema ?? {})
  const flagKeys = Object.keys(toolkit?.flags ?? {})

  const switchMode = (next: ArgMode) => {
    if (next === 'literal') onChange(0)
    else if (next === 'param') onChange({ param: paramKeys[0] ?? '' })
    else onChange({ flag: flagKeys[0] ?? '' })
  }

  const btnStyle = (active: boolean): React.CSSProperties => ({
    padding: '2px 8px',
    fontSize: '11px',
    borderRadius: '4px',
    border: '1px solid var(--surface2)',
    background: active ? 'var(--surface2)' : 'transparent',
    color: active ? 'var(--text)' : 'var(--subtext0)',
    cursor: 'pointer',
  })

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: '4px' }}>
      <div style={{ display: 'flex', gap: '4px' }}>
        {(['literal', 'param', 'flag'] as ArgMode[]).map(m => (
          <button key={m} style={btnStyle(mode === m)} onClick={() => switchMode(m)}>
            {m.charAt(0).toUpperCase() + m.slice(1)}
          </button>
        ))}
      </div>

      {mode === 'literal' && (
        <input
          type="number"
          value={typeof value === 'number' ? value : 0}
          onChange={e => onChange(Number(e.target.value))}
          style={{ width: '100%' }}
        />
      )}

      {mode === 'param' && (
        paramKeys.length > 0 ? (
          <select
            value={(value as { param: string }).param}
            onChange={e => onChange({ param: e.target.value })}
            style={{ width: '100%' }}
          >
            {paramKeys.map(k => <option key={k} value={k}>{k}</option>)}
          </select>
        ) : (
          <input
            type="text"
            value={(value as { param: string }).param ?? ''}
            onChange={e => onChange({ param: e.target.value })}
            placeholder="param name"
            style={{ width: '100%' }}
          />
        )
      )}

      {mode === 'flag' && (
        flagKeys.length > 0 ? (
          <select
            value={(value as { flag: string }).flag}
            onChange={e => onChange({ flag: e.target.value })}
            style={{ width: '100%' }}
          >
            {flagKeys.map(k => <option key={k} value={k}>{k}</option>)}
          </select>
        ) : (
          <input
            type="text"
            value={(value as { flag: string }).flag ?? ''}
            onChange={e => onChange({ flag: e.target.value })}
            placeholder="flag name"
            style={{ width: '100%' }}
          />
        )
      )}
    </div>
  )
}

import type { FdaState, FdaAction, ToolkitRead } from '../types'
import ActionEditor from './ActionEditor'

const DEFAULT_ACTION: FdaAction = { type: 'hardware', ref: '', method: 'set', args: [1] }

interface Props {
  stateName: string
  state: FdaState
  toolkit: ToolkitRead | null
  onChange: (updated: FdaState) => void
}

export default function StateBodyPanel({ stateName, state, toolkit, onChange }: Props) {
  const isPassthrough = !state.entry_actions?.length && (toolkit?.states?.includes(stateName) ?? false)
  const actions = state.entry_actions ?? []

  const updateAction = (i: number, updated: FdaAction) => {
    const newActions = actions.map((a, idx) => idx === i ? updated : a)
    onChange({ ...state, entry_actions: newActions })
  }

  const removeAction = (i: number) => {
    const newActions = actions.filter((_, idx) => idx !== i)
    onChange({ ...state, entry_actions: newActions })
  }

  const addAction = () => {
    onChange({ ...state, entry_actions: [...actions, { ...DEFAULT_ACTION }] })
  }

  return (
    <div>
      <div style={{ fontSize: '12px', color: 'var(--subtext0)', marginBottom: '8px', fontWeight: 500 }}>
        STATE BODY
      </div>
      <div style={{ fontSize: '13px', fontWeight: 600, fontFamily: 'monospace', marginBottom: '12px', color: 'var(--text)' }}>
        {stateName}
        {isPassthrough && (
          <span style={{ marginLeft: '8px', fontSize: '11px', color: '#9ca3af', fontFamily: 'sans-serif', fontWeight: 400 }}>
            🔒 passthrough
          </span>
        )}
      </div>

      {isPassthrough ? (
        <p style={{ fontSize: '12px', color: 'var(--subtext0)', margin: 0 }}>
          This state delegates to the Python method of the same name. Add an entry_action to override.
        </p>
      ) : (
        <>
          {actions.length === 0 ? (
            <p style={{ fontSize: '12px', color: 'var(--subtext0)', marginBottom: '12px' }}>
              No entry actions. Add one below.
            </p>
          ) : (
            <div style={{ display: 'flex', flexDirection: 'column', gap: '8px', marginBottom: '12px' }}>
              {actions.map((action, i) => (
                <div key={i} style={{ position: 'relative' }}>
                  <ActionEditor action={action} toolkit={toolkit} onChange={updated => updateAction(i, updated)} />
                  <button
                    className="button-danger"
                    style={{ fontSize: '11px', padding: '2px 6px', marginTop: '4px' }}
                    onClick={() => removeAction(i)}
                  >
                    Remove
                  </button>
                </div>
              ))}
            </div>
          )}

          <button className="button-secondary" style={{ fontSize: '12px', width: '100%' }} onClick={addAction}>
            + Add action
          </button>
        </>
      )}

      {state.wait_condition && (
        <div style={{ marginTop: '16px', padding: '8px', background: 'var(--surface0)', borderRadius: '6px' }}>
          <div style={{ fontSize: '11px', color: 'var(--subtext0)', marginBottom: '4px' }}>wait_condition (read-only)</div>
          <code style={{ fontSize: '12px', color: '#60a5fa' }}>
            {state.wait_condition.view} {state.wait_condition.op} {String(state.wait_condition.rhs)}
          </code>
        </div>
      )}
    </div>
  )
}

import { useState } from 'react'
import type { ToolkitRead } from '../types'

interface HwOverride {
  group: string
  id: string
}

interface Props {
  overrides: Record<string, unknown>
  toolkit: ToolkitRead | null
  onChange: (updated: Record<string, unknown>) => void
}

export default function HwOverridesPanel({ overrides, toolkit, onChange }: Props) {
  const [expanded, setExpanded] = useState(false)

  const hwKeys = Object.keys(toolkit?.semantic_hardware ?? {})
  if (hwKeys.length === 0) return null

  const getOverride = (key: string): HwOverride => {
    const existing = overrides[key]
    if (existing && typeof existing === 'object') {
      return { group: (existing as HwOverride).group ?? '', id: (existing as HwOverride).id ?? '' }
    }
    return { group: '', id: '' }
  }

  const updateKey = (key: string, field: 'group' | 'id', val: string) => {
    const current = getOverride(key)
    const updated = { ...current, [field]: val }
    // If both fields empty, remove the override
    if (!updated.group && !updated.id) {
      const { [key]: _, ...rest } = overrides
      onChange(rest)
    } else {
      onChange({ ...overrides, [key]: updated })
    }
  }

  const activeCount = hwKeys.filter(k => overrides[k]).length

  return (
    <div style={{ borderTop: '1px solid var(--surface1)', paddingTop: '12px', marginTop: '12px' }}>
      <button
        className="button-link"
        style={{ fontSize: '12px', display: 'flex', alignItems: 'center', gap: '6px', padding: 0 }}
        onClick={() => setExpanded(v => !v)}
      >
        {expanded ? '▾' : '▸'} Hardware Overrides
        {activeCount > 0 && (
          <span className="badge" style={{ fontSize: '10px' }}>{activeCount}</span>
        )}
      </button>

      {expanded && (
        <div style={{ marginTop: '10px', display: 'flex', flexDirection: 'column', gap: '10px' }}>
          <p style={{ fontSize: '11px', color: 'var(--subtext0)', margin: 0 }}>
            Override toolkit SEMANTIC_HARDWARE for this task definition only.
          </p>
          {hwKeys.map(key => {
            const override = getOverride(key)
            return (
              <div key={key}>
                <div style={{ fontSize: '12px', color: 'var(--text)', marginBottom: '4px', fontFamily: 'monospace' }}>
                  {key}
                </div>
                <div style={{ display: 'flex', gap: '6px' }}>
                  <input
                    type="text"
                    placeholder="group"
                    value={override.group}
                    onChange={e => updateKey(key, 'group', e.target.value)}
                    style={{ flex: 1 }}
                  />
                  <input
                    type="text"
                    placeholder="id"
                    value={override.id}
                    onChange={e => updateKey(key, 'id', e.target.value)}
                    style={{ flex: 1 }}
                  />
                </div>
              </div>
            )
          })}
        </div>
      )}
    </div>
  )
}

**stripClientFields helper** (add before the return):
```tsx
function stripClientFields(fda: FdaJson): FdaJson {
  const cleanStates = Object.fromEntries(
    Object.entries(fda.states).map(([name, state]) => {
      const { _passthrough, ...rest } = state as typeof state & { _passthrough?: boolean }
      return [name, rest]
    })
  )
  return { ...fda, states: cleanStates }
}
```
