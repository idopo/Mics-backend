**Requirements:** HW-01, HW-02, HW-03, HW-04, HW-05

## Context

Hardware driver classes (gpio.py, i2c.py, etc.) currently live exclusively on the Pi. This plan centralizes the driver source in the backend DB, extracts AST metadata for later GUI use, and proves end-to-end delivery to the Pi via `LOAD_HARDWARE_LIBS` ZMQ. It also establishes a lifecycle: every edit creates a new version; versions graduate through `unvalidated → beta → stable`; all stable versions are preserved for rollback.

gpio.py is the Phase 9 target. i2c.py has late imports and an optional MLX90640 class removal — not blocking.

## Lib Lifecycle States

```
unvalidated  →  beta  →  stable
```

| State | Meaning | How reached |
|---|---|---|
| `unvalidated` | Upload failed Pi import test, or Pi unavailable | Default if AST fails; or Pi test failure |
| `beta` | Server AST passed; deployable but not hardware-confirmed | Server-side AST check passes; OR Pi import test passes |
| `stable` | Confirmed working on real hardware | Auto: `INC_TRIAL_COUNTER` in a session using this lib; OR user calls `mark-stable` |

On upload: server AST + py_compile → if passes → state=`beta` immediately (deployable without Pi test).
If `TEST_PILOT` configured: orchestrator also runs Pi import test; if that fails, state → `unvalidated` + traceback stored.
Orchestrator deploys any lib with state `beta` or `stable`. `unvalidated` → skipped with warning.

```python
def run_task_definition_toolkit_id_migration(engine):
    """Phase 09: add task_definitions.toolkit_id (FK → task_toolkits.id) and
    backfill it from the legacy toolkit_name string column. Idempotent."""
    with engine.begin() as conn:  # auto-commits on success, rolls back on exception
        conn.execute(text(
            "ALTER TABLE task_definitions "
            "ADD COLUMN IF NOT EXISTS toolkit_id INTEGER REFERENCES task_toolkits(id)"
        ))
        # Backfill: pick the canonical toolkit first, then oldest with matching name.
        # Only updates rows where toolkit_id is still NULL and toolkit_name is set.
        conn.execute(text("""
            UPDATE task_definitions td
            SET toolkit_id = (
                SELECT tt.id FROM task_toolkits tt
                WHERE tt.name = td.toolkit_name
                ORDER BY tt.is_canonical DESC, tt.created_at ASC
                LIMIT 1
            )
            WHERE td.toolkit_name IS NOT NULL AND td.toolkit_id IS NULL
        """))
```

The `IF NOT EXISTS` clause makes the ALTER idempotent. The backfill only touches rows where `toolkit_id IS NULL`, so re-running the migration is a no-op once data is in place.

If the legacy `toolkit_name` column does not exist in some environments, the UPDATE will raise `UndefinedColumn`. Guard with a check:

```python
def run_task_definition_toolkit_id_migration(engine):

            SELECT 1 FROM information_schema.columns
            WHERE table_name = 'task_definitions' AND column_name = 'toolkit_name'
```

### Step 2: AST extraction + source validation utilities

**File:** `api/routers/hardware_libs.py` (new file)

```python
import ast, hashlib, py_compile, tempfile, os

def extract_ast_metadata(source_code: str) -> dict:
    """Return {classes: [{name, methods: [{name, args: [{name, annotation, default}]}]}]}.
    Includes __init__ so Phase 10 can derive constructor args for hardware config fields.
    """
    tree = ast.parse(source_code)
    classes = []
    for node in ast.walk(tree):
        if isinstance(node, ast.ClassDef):
            methods = []
            for item in node.body:
                if isinstance(item, ast.FunctionDef):
                    args = []
                    for arg in item.args.args:
                        if arg.arg == "self":
                            continue
                        annotation = ast.unparse(arg.annotation) if arg.annotation else None
                        args.append({"name": arg.arg, "annotation": annotation})
                    defaults = item.args.defaults
                    for i, default in enumerate(defaults):
                        args[len(args) - len(defaults) + i]["default"] = ast.unparse(default)
                    methods.append({"name": item.name, "args": args})
            classes.append({"name": node.name, "methods": methods})
    return {"classes": classes}

def validate_source(source_code: str, filename: str = "<string>") -> tuple[bool, str | None]:
    """Returns (ok, error). AST parse + py_compile for full check."""
    try:
        ast.parse(source_code, filename=filename)
    except SyntaxError as e:
        return False, f"SyntaxError at line {e.lineno}: {e.msg}"
    with tempfile.NamedTemporaryFile(suffix=".py", delete=False) as f:
        f.write(source_code.encode())
        tmp = f.name
    try:
        py_compile.compile(tmp, doraise=True)
        return True, None
    except py_compile.PyCompileError as e:
        return False, str(e)
    finally:
        os.unlink(tmp)

def sha256(source_code: str) -> str:
    return hashlib.sha256(source_code.encode()).hexdigest()
```

```
GET /api/hardware-libs/{lib_id}/versions/diff?from={version_id_1}&to={version_id_2}
    → {
        removed_methods:    [{class_name, method_name}],
        changed_signatures: [{class_name, method_name, old_args, new_args}],
        added_methods:      [{class_name, method_name}]
      }
```

Uses stored `ast_metadata` from both `hardware_lib_versions` rows. If `ast_metadata` is null for either row (pre-migration record), parse `source_code` on-demand using `extract_ast_metadata()` from Step 2.

Diff logic:
```python
def diff_ast(old_meta: dict, new_meta: dict) -> dict:
    removed, changed, added = [], [], []
    old_classes = {c["name"]: {m["name"]: m for m in c["methods"]} for c in old_meta.get("classes", [])}
    new_classes = {c["name"]: {m["name"]: m for m in c["methods"]} for c in new_meta.get("classes", [])}
    for cls_name, old_methods in old_classes.items():
        new_methods = new_classes.get(cls_name, {})
        for method_name, old_m in old_methods.items():
            if method_name not in new_methods:
                removed.append({"class_name": cls_name, "method_name": method_name})
            elif old_m["args"] != new_methods[method_name]["args"]:
                changed.append({"class_name": cls_name, "method_name": method_name,
                                 "old_args": old_m["args"], "new_args": new_methods[method_name]["args"]})
        for method_name in new_methods:
            if method_name not in old_methods:
                added.append({"class_name": cls_name, "method_name": method_name})
    return {"removed_methods": removed, "changed_signatures": changed, "added_methods": added}
```

```python
def l_load_hardware_libs(self, value):
    """Write received hardware lib files to override dir; optionally run import test."""
    libs = value.get("libs", [])
    test_import = value.get("test_import", False)
    version_id = value.get("version_id")

    if not libs:
        return
```

```python
from pathlib import Path

HARDWARE_OVERRIDE_DIR = Path.home() / "apps" / "hardware_overrides"

def _write_hardware_libs(libs: list[dict]) -> tuple[bool, str | None]:
    """Write libs to override dir and add dir to sys.path. Returns (ok, error)."""
    HARDWARE_OVERRIDE_DIR.mkdir(parents=True, exist_ok=True)
    override_str = str(HARDWARE_OVERRIDE_DIR)
    if override_str not in sys.path:

        dest.write_text(lib["source_code"])

        if module_name in sys.modules:
            importlib.reload(sys.modules[module_name])
    return True, None
```

#### 7h. API helpers (`web_ui/react-src/src/api/hardware_libs.ts`)

```ts
import { apiFetch } from "./client";
import type { HardwareLib, HardwareLibVersion } from "../types";

export async function listHardwareLibs(): Promise<HardwareLib[]> {
  return apiFetch("/api/hardware-libs");
}
export async function getHardwareLib(id: number): Promise<HardwareLib> {
  return apiFetch(`/api/hardware-libs/${id}`);
}
export async function uploadHardwareLib(name: string, file: File): Promise<HardwareLib> {
  const fd = new FormData();
  fd.append("name", name);
  fd.append("file", file);
  return apiFetch("/api/hardware-libs", { method: "POST", body: fd });
}
export async function updateHardwareLibSource(id: number, source_code: string): Promise<HardwareLib> {
  return apiFetch(`/api/hardware-libs/${id}`, {
    method: "PUT",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ source_code }),
  });
}
export async function listVersions(id: number, state?: string): Promise<HardwareLibVersion[]> {
  const q = state ? `?state=${encodeURIComponent(state)}` : "";
  return apiFetch(`/api/hardware-libs/${id}/versions${q}`);
}
export async function markStable(id: number): Promise<HardwareLib> {
  return apiFetch(`/api/hardware-libs/${id}/mark-stable`, { method: "PATCH" });
}
export async function rollback(id: number, version_id: number): Promise<HardwareLib> {
  return apiFetch(`/api/hardware-libs/${id}/rollback`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ version_id }),
  });
}
```

**Required: update `web_ui/react-src/src/api/client.ts`.** The current `apiFetch` unconditionally merges `'Content-Type': 'application/json'` into every request. This breaks multipart uploads because the browser cannot set the correct `multipart/form-data; boundary=...` header when `Content-Type` is already set. Fix by skipping the default header when body is `FormData`:

```ts
const headers: Record<string, string> = {};
if (!(options?.body instanceof FormData)) {
  headers['Content-Type'] = 'application/json';
}
// merge with any caller-supplied headers
```

This is a global change to `client.ts` — verify no other existing callers pass `FormData` without already overriding Content-Type (none currently do).
