@app.post("/push-fda")
def push_fda(body: dict):
    """
    Push UPDATE_FDA ZMQ message to a named pilot.
    body: {pilot_name: str, fda_json: dict}
    """
    pilot_name = body.get("pilot_name")
    fda_json = body.get("fda_json")
    if not pilot_name or fda_json is None:
        raise HTTPException(400, "pilot_name and fda_json are required")
    try:
        station.push_hot_reload(pilot_name, fda_json)
        return {"ok": True, "pilot": pilot_name}
    except ValueError as e:
        raise HTTPException(400, detail=str(e))
    except Exception as e:
        raise HTTPException(500, detail=str(e))
```

import os as _os
import requests as _requests

ORCHESTRATOR_URL = _os.getenv("ORCHESTRATOR_URL", "http://orchestrator:9000")

@app.post("/task-definitions/{defn_id}/push")
def push_task_definition(
    defn_id: int,
    pilot: str,  # query param: ?pilot=T
    _: dict = Depends(verify_token),
):
    """
    Validate fda_json against toolkit, then forward UPDATE_FDA to orchestrator.
    """
    db: OrmSession = SA_SessionLocal()
    try:
        # 1. Load task definition (use raw SQL for new columns)
        row = db.execute(sa_text(
            "SELECT id, task_name, toolkit_name, fda_json FROM task_definitions WHERE id = :id"
        ), {"id": defn_id}).fetchone()
        if not row:
            raise HTTPException(404, "Task definition not found")

        fda_json = json.loads(row.fda_json) if isinstance(row.fda_json, str) else row.fda_json
        if fda_json is None:
            raise HTTPException(400, "Task definition has no fda_json")

        # 2. Validate fda_json against toolkit (warn-only if toolkit missing)
        validation_errors = []
        if row.toolkit_name:
            toolkit = (
                db.query(TaskToolkit)
                .filter(TaskToolkit.name == row.toolkit_name)
                .order_by(TaskToolkit.updated_at.desc())
                .first()
            )
            if toolkit:
                validation_errors = _validate_fda_against_toolkit(fda_json, toolkit)
            else:
                # Toolkit not yet in DB (Pi hasn't connected) — skip validation
                pass

        if validation_errors:
            raise HTTPException(422, detail={"errors": validation_errors})

        # 3. Forward to orchestrator /push-fda
        try:
            resp = _requests.post(
                f"{ORCHESTRATOR_URL}/push-fda",
                json={"pilot_name": pilot, "fda_json": fda_json},
                timeout=5,
            )
            resp.raise_for_status()
        except _requests.exceptions.ConnectionError:
            raise HTTPException(503, "Orchestrator not reachable")
        except _requests.exceptions.HTTPError as e:
            raise HTTPException(502, f"Orchestrator error: {e.response.text}")

        return {"status": "pushed", "pilot": pilot, "task_definition_id": defn_id}

    finally:
        db.close()


def _validate_fda_against_toolkit(fda_json: dict, toolkit) -> list:
    """
    Returns list of error strings. Empty list = valid.
    Validates:
    - All state names in fda_json.states keys exist in toolkit.states
    - All semantic_hardware refs in entry_actions/conditions exist in toolkit.semantic_hardware
    """
    errors = []
    known_states = set(toolkit.states or [])
    known_hw = set((toolkit.semantic_hardware or {}).keys())

    fda_states = fda_json.get("states", {})
    if isinstance(fda_states, dict):
        for state_name in fda_states:
            if known_states and state_name not in known_states:
                errors.append(f"Unknown state: '{state_name}' not in toolkit states")

        for state_name, state_body in fda_states.items():
            for action in (state_body.get("entry_actions") or []):
                ref = action.get("ref")
                action_type = action.get("type")
                if action_type == "hardware" and ref and known_hw and ref not in known_hw:
                    errors.append(
                        f"State '{state_name}': unknown hardware ref '{ref}'"
                    )

    return errors
```
