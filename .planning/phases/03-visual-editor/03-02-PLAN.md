import { Handle, Position, type NodeProps } from '@xyflow/react'
import type { FdaState, ToolkitRead } from '../types'

export type StateNodeData = {
  name: string
  state: FdaState
  isInitial: boolean
  toolkit: ToolkitRead | null
  onSelect?: (name: string) => void
}

// Passthrough = no entry_actions defined AND state name appears in toolkit.states
function isPassthrough(name: string, state: FdaState, toolkit: ToolkitRead | null): boolean {
  if (state.entry_actions && state.entry_actions.length > 0) return false
  if (!toolkit?.states) return false
  return toolkit.states.includes(name)
}

export default function StateNode({ data, selected }: NodeProps) {
  const { name, state, isInitial, toolkit, onSelect } = data as StateNodeData
  const passthrough = isPassthrough(name, state, toolkit)

  const nodeColor = isInitial ? '#7c3aed' : passthrough ? '#6b7280' : '#2563eb'
  const actionCount = state.entry_actions?.length ?? 0

  return (
    <div
      onClick={() => onSelect?.(name)}
      style={{
        background: '#1e2130',
        border: '1px solid #2d3148',
        borderLeft: `4px solid ${nodeColor}`,
        borderRadius: '8px',
        padding: '10px 14px',
        minWidth: '180px',
        cursor: 'pointer',
        boxShadow: selected
          ? `0 0 0 2px ${nodeColor}, 0 0 16px color-mix(in srgb, ${nodeColor} 40%, transparent)`
          : `0 0 12px color-mix(in srgb, ${nodeColor} 20%, transparent)`,
        color: '#e2e8f0',
        fontFamily: 'monospace',
        fontSize: '13px',
        userSelect: 'none',
      }}
    >
      <Handle type="target" position={Position.Top} style={{ background: nodeColor }} />

      <div style={{ fontWeight: 600, marginBottom: '4px', letterSpacing: '0.03em' }}>
        {name}
        {isInitial && (
          <span style={{ marginLeft: '6px', fontSize: '10px', color: '#a78bfa', verticalAlign: 'middle' }}>
            INIT
          </span>
        )}
      </div>

      <div style={{ display: 'flex', gap: '6px', alignItems: 'center' }}>
        {passthrough ? (
          <span style={{ fontSize: '11px', color: '#9ca3af' }}>
            🔒 {'{py}'}

import type { FdaCondition, ToolkitRead } from '../types'

const OPS = ['==', '!=', '>=', '<=', '>', '<'] as const

interface Props {
  condition: FdaCondition
  toolkit: ToolkitRead | null
  onChange: (updated: FdaCondition) => void
}

// Build combined list of view keys: semantic_hardware keys + flags keys + params_schema keys
function buildViewOptions(toolkit: ToolkitRead | null): string[] {
  const opts: string[] = []
  if (toolkit?.semantic_hardware) opts.push(...Object.keys(toolkit.semantic_hardware))
  if (toolkit?.flags) opts.push(...Object.keys(toolkit.flags))
  if (toolkit?.params_schema) opts.push(...Object.keys(toolkit.params_schema))
  return [...new Set(opts)]
}

export default function ConditionBuilder({ condition, toolkit, onChange }: Props) {
  const viewOptions = buildViewOptions(toolkit)

  const update = (patch: Partial<FdaCondition>) =>
    onChange({ ...condition, ...patch })

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: '12px' }}>
      <div>
        <label style={{ fontSize: '11px', color: 'var(--subtext0)', display: 'block', marginBottom: '4px' }}>
          View key
        </label>
        {viewOptions.length > 0 ? (
          <select
            value={condition.view}
            onChange={e => update({ view: e.target.value })}
            style={{ width: '100%' }}
          >
            {!viewOptions.includes(condition.view) && (
              <option value={condition.view}>{condition.view}</option>
            )}
            {viewOptions.map(k => (
              <option key={k} value={k}>{k}</option>
            ))}
          </select>
        ) : (
          <input
            type="text"
            value={condition.view}
            onChange={e => update({ view: e.target.value })}
            placeholder="e.g. licker1"
            style={{ width: '100%' }}
          />
        )}
      </div>

      <div>
        <label style={{ fontSize: '11px', color: 'var(--subtext0)', display: 'block', marginBottom: '4px' }}>
          Operator
        </label>
        <select
          value={condition.op}
          onChange={e => update({ op: e.target.value as FdaCondition['op'] })}
          style={{ width: '100%' }}
        >
          {OPS.map(op => <option key={op} value={op}>{op}</option>)}
        </select>
      </div>

      <div>
        <label style={{ fontSize: '11px', color: 'var(--subtext0)', display: 'block', marginBottom: '4px' }}>
          Right-hand value
        </label>
        <input
          type="text"
          value={String(condition.rhs ?? '')}
          onChange={e => {
            // Attempt numeric parse; fall back to string
            const raw = e.target.value
            const num = Number(raw)
            update({ rhs: raw !== '' && !isNaN(num) ? num : raw })
          }}
          placeholder="e.g. 1"
          style={{ width: '100%' }}
        />
      </div>
    </div>
  )
}

```tsx
import { useState, useEffect, useCallback } from 'react'
import { useParams, Link } from 'react-router-dom'
import { useQuery } from '@tanstack/react-query'
import {
  ReactFlow,
  Background,
  Controls,
  MiniMap,
  useNodesState,
  useEdgesState,
  addEdge,
  type Node,
  type Edge,
  type Connection,
} from '@xyflow/react'
import '@xyflow/react/dist/style.css'

import { getTaskDefinition } from '../../api/task-definitions'
import { getToolkitsByName } from '../../api/toolkits'
import type { FdaJson, FdaTransition, ToolkitRead } from '../../types'
import StateNode from '../../components/StateNode'
import ConditionBuilder from '../../components/ConditionBuilder'

const nodeTypes = { stateNode: StateNode }

function fdaToNodes(fdaJson: FdaJson, toolkit: ToolkitRead | null, onSelect: (name: string) => void): Node[] {
  return Object.entries(fdaJson.states).map(([name, state], i) => ({
    id: name,
    type: 'stateNode',
    position: { x: (i % 4) * 270, y: Math.floor(i / 4) * 170 },
    data: {
      name,
      state,
      isInitial: name === fdaJson.initial_state,
      toolkit,
      onSelect,
    },
  }))
}

function fdaToEdges(fdaJson: FdaJson): Edge[] {
  return fdaJson.transitions.map((t: FdaTransition, i: number) => ({
    id: `e-${i}`,
    source: t.from_state,
    target: t.next_state,
    label: `${t.condition.view} ${t.condition.op} ${t.condition.rhs}`,
    data: { transition: t },
    style: { stroke: '#475569' },
    labelStyle: { fill: '#94a3b8', fontSize: 11 },
    labelBgStyle: { fill: '#1e2130' },
  }))
}

export default function TaskEditor() {
  const { id } = useParams<{ id: string }>()
  const numId = Number(id)

  const { data: taskDef, isLoading } = useQuery({
    queryKey: ['task-definition', numId],
    queryFn: () => getTaskDefinition(numId),
    enabled: !isNaN(numId),
  })

  const { data: toolkits } = useQuery({
    queryKey: ['toolkits-by-name', taskDef?.toolkit_name],
    queryFn: () => getToolkitsByName(taskDef!.toolkit_name!),
    enabled: !!taskDef?.toolkit_name,
  })

  // Use the first (or only) toolkit variant for metadata
  const toolkit = toolkits?.[0] ?? null
  const hasMultipleVariants = (toolkits?.length ?? 0) > 1

  const [fdaJson, setFdaJson] = useState<FdaJson | null>(null)
  const [selectedState, setSelectedState] = useState<string | null>(null)
  const [selectedEdge, setSelectedEdge] = useState<Edge | null>(null)

  const [nodes, setNodes, onNodesChange] = useNodesState([])
  const [edges, setEdges, onEdgesChange] = useEdgesState([])

  // Initialize fdaJson from query data
  useEffect(() => {
    if (taskDef?.fda_json) {
      setFdaJson(taskDef.fda_json)
    }
  }, [taskDef])

  // Sync canvas whenever fdaJson or toolkit changes
  useEffect(() => {
    if (!fdaJson) return
    setNodes(fdaToNodes(fdaJson, toolkit, (name) => {
      setSelectedState(name)
      setSelectedEdge(null)
    }))
    setEdges(fdaToEdges(fdaJson))
  }, [fdaJson, toolkit])

  // Drag-to-connect: creates a new edge and appends a transition to fdaJson
  const onConnect = useCallback((params: Connection) => {
    setEdges(eds => addEdge({ ...params, label: 'new condition' }, eds))
    setFdaJson(prev => prev ? {
      ...prev,
      transitions: [...prev.transitions, {
        from_state: params.source!,
        condition: { view: '', op: '==', rhs: 0 },
        next_state: params.target!,
      }],
    } : prev)
  }, [setEdges])

  const updateTransitionCondition = (edgeId: string, updated: FdaJson['transitions'][number]['condition']) => {
    if (!fdaJson) return
    const idx = parseInt(edgeId.replace('e-', ''), 10)
    const newTransitions = fdaJson.transitions.map((t, i) =>
      i === idx ? { ...t, condition: updated } : t
    )
    const newFda = { ...fdaJson, transitions: newTransitions }
    setFdaJson(newFda)
    // Also update selected edge label
    if (selectedEdge) {
      setSelectedEdge(prev => prev ? {
        ...prev,
        label: `${updated.view} ${updated.op} ${updated.rhs}`,
        data: { transition: newTransitions[idx] },
      } : null)
    }
  }

  const displayName = taskDef?.display_name ?? taskDef?.task_name ?? `Task #${id}`

  return (
    <div style={{ display: 'flex', flexDirection: 'column', height: '100vh' }}>
      {/* Header */}
      <div style={{
        display: 'flex',
        alignItems: 'center',
        gap: '12px',
        padding: '10px 16px',
        background: 'var(--surface0)',
        borderBottom: '1px solid var(--surface1)',
        flexShrink: 0,
      }}>
        <Link to="/task-definitions-ui" className="button-link" style={{ fontSize: '13px' }}>
          ← Tasks
        </Link>
        <h2 style={{ margin: 0, fontSize: '16px', flex: 1 }}>{displayName}</h2>
        {taskDef?.toolkit_name && (
          <span className="badge">{taskDef.toolkit_name}</span>
        )}
      </div>

      {/* Variant warning banner */}
      {hasMultipleVariants && (
        <div style={{
          background: 'rgba(234, 179, 8, 0.08)',
          borderBottom: '1px solid rgba(234, 179, 8, 0.25)',
          padding: '6px 16px',
          fontSize: '12px',
          color: '#ca8a04',
          flexShrink: 0,
        }}>
          ⚠ This toolkit has {toolkits!.length} hardware variants. Binding to a specific variant is available in a future update.
        </div>
      )}

      {/* Body: canvas + right panel */}
      <div style={{ display: 'flex', flex: 1, overflow: 'hidden' }}>
        {/* Canvas */}
        <div style={{ flex: 1, background: '#0d1117' }}>
          {isLoading ? (
            <div style={{ color: '#94a3b8', padding: '2rem', fontSize: '14px' }}>Loading…</div>
          ) : !fdaJson ? (
            <div style={{ color: '#94a3b8', padding: '2rem', fontSize: '14px' }}>
              No FDA JSON stored for this task definition yet.
            </div>
          ) : (
            <ReactFlow
              nodes={nodes}
              edges={edges}
              nodeTypes={nodeTypes}
              onNodesChange={onNodesChange}
              onEdgesChange={onEdgesChange}
              onConnect={onConnect}
              onEdgeClick={(_evt, edge) => {
                setSelectedEdge(edge)
                setSelectedState(null)
              }}
              onPaneClick={() => {
                setSelectedEdge(null)
                setSelectedState(null)
              }}
              fitView
            >
              <Background color="#1e2130" gap={20} />
              <Controls />
              <MiniMap nodeColor={() => '#2563eb'} style={{ background: '#1e2130' }} />
            </ReactFlow>
          )}
        </div>

        {/* Right panel (320px) */}
        <div style={{
          width: '320px',
          flexShrink: 0,
          background: 'var(--base)',
          borderLeft: '1px solid var(--surface1)',
          padding: '16px',
          overflowY: 'auto',
          display: 'flex',
          flexDirection: 'column',
          gap: '12px',
        }}>
          {selectedEdge && fdaJson ? (() => {
            const idx = parseInt(selectedEdge.id.replace('e-', ''), 10)
            const transition = fdaJson.transitions[idx]
            if (!transition) return null
            return (
              <div>
                <div style={{ fontSize: '12px', color: 'var(--subtext0)', marginBottom: '10px', fontWeight: 500 }}>
                  TRANSITION CONDITION
                </div>
                <div style={{ fontSize: '12px', color: 'var(--subtext1)', marginBottom: '12px' }}>
                  {transition.from_state} → {transition.next_state}
                </div>
                <ConditionBuilder
                  condition={transition.condition}
                  toolkit={toolkit}
                  onChange={updated => updateTransitionCondition(selectedEdge.id, updated)}
                />
              </div>
            )
          })() : selectedState ? (
            <div>
              <div style={{ fontSize: '12px', color: 'var(--subtext0)', marginBottom: '10px', fontWeight: 500 }}>
                STATE BODY
              </div>
              <div style={{ fontSize: '13px', color: 'var(--text)', marginBottom: '8px', fontWeight: 600, fontFamily: 'monospace' }}>
                {selectedState}
              </div>
              <p style={{ fontSize: '12px', color: 'var(--subtext0)', margin: 0 }}>
                State body editing available in next update.
              </p>
            </div>
          ) : (
            <div style={{ color: 'var(--subtext0)', fontSize: '13px', marginTop: '2rem', textAlign: 'center' }}>
              Select a state or transition to inspect
            </div>
          )}
        </div>
      </div>
    </div>
  )
}
```

Key implementation notes:
- `onConnect` uses `useCallback` (imported from react) to avoid stale closure issues with `setEdges`.
- The new transition appended by `onConnect` gets `condition: { view: '', op: '==', rhs: 0 }` as a default — the user must click the new edge to fill in the real condition via ConditionBuilder.
- `addEdge` and `Connection` are imported from `@xyflow/react`.
- The `onSelect` callback is passed through node `data` so StateNode can call it on click — this avoids the need for a global event bus or context.
- Edge condition updates patch the in-memory fdaJson via `updateTransitionCondition`; these changes are NOT saved until Plan 03-03 adds the Save button.
- No `blocking` field is used anywhere — states only have `wait_condition`.
- `fitView` on ReactFlow ensures all nodes are visible on load regardless of position.
- The right panel state-body placeholder message is intentional — full editing comes in Plan 03-03.
