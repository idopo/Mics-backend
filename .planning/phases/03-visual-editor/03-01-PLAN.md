  type: 'hardware' | 'flag' | 'timer' | 'special' | 'method'
  ref?: string          // semantic hw key / flag name / callable method name
  method?: string       // e.g. "set", "toggle", "increment", "reset"
  args?: unknown[]      // each arg: literal scalar | { param: string } | { flag: string }

  duration?: unknown    // for type=timer: literal ms value or { param: string }
}

export interface FdaState {

  return_data?: unknown[]

```tsx
import { useQuery } from '@tanstack/react-query'
import { useNavigate } from 'react-router-dom'
import { getTaskDefinitions } from '../../api/task-definitions'
import { getToolkits } from '../../api/toolkits'
import type { TaskDefinitionFull, ToolkitRead } from '../../types'

// Build a set of toolkit names that have 2+ hw_hash variants
function buildVariantNames(toolkits: ToolkitRead[]): Set<string> {
  const counts: Record<string, number> = {}
  for (const t of toolkits) {
    counts[t.name] = (counts[t.name] ?? 0) + 1
  }
  return new Set(Object.entries(counts).filter(([, n]) => n >= 2).map(([name]) => name))
}

export default function TaskDefinitions() {
  const navigate = useNavigate()

  const { data: taskDefs, isLoading: loadingDefs } = useQuery({
    queryKey: ['task-definitions'],
    queryFn: getTaskDefinitions,
  })

  const { data: toolkits } = useQuery({
    queryKey: ['toolkits'],
    queryFn: getToolkits,
  })

  const variantNames = toolkits ? buildVariantNames(toolkits) : new Set<string>()

  const formatDate = (iso: string) =>
    new Date(iso).toLocaleDateString(undefined, { year: 'numeric', month: 'short', day: 'numeric' })

  return (
    <div className="container">
      <section className="card">
        <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '1rem' }}>
          <h2 style={{ margin: 0 }}>Task Definitions</h2>
        </div>

        {loadingDefs ? (
          <p style={{ color: 'var(--subtext0)', fontSize: '14px' }}>Loading…</p>
        ) : (taskDefs ?? []).length === 0 ? (
          <p style={{ color: 'var(--subtext0)', fontSize: '14px' }}>No task definitions yet.</p>
        ) : (
          <ul className="scroll-list" style={{ listStyle: 'none', padding: 0, margin: 0 }}>
            {(taskDefs ?? []).map((def: TaskDefinitionFull) => (
              <li
                key={def.id}
                className="subject-item fade-in-item"
                style={{ cursor: 'pointer', padding: '10px 12px', borderBottom: '1px solid var(--surface1)' }}
                onClick={() => navigate(`/task-editor/${def.id}`)}
              >
                <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'flex-start' }}>
                  <div>
                    <div style={{ fontWeight: 500, fontSize: '14px' }}>
                      {def.display_name ?? def.task_name}
                    </div>
                    <div style={{ display: 'flex', gap: '6px', marginTop: '4px', alignItems: 'center' }}>
                      {def.toolkit_name && (
                        <span className="badge">
                          {def.toolkit_name}
                        </span>
                      )}
                      {def.toolkit_name && variantNames.has(def.toolkit_name) && (
                        <span
                          className="badge"
                          style={{ background: 'rgba(234, 179, 8, 0.15)', color: '#ca8a04', border: '1px solid rgba(234, 179, 8, 0.3)' }}
                          title="Multiple hardware variants exist for this toolkit"
                        >
                          ⚠ 2+ variants
                        </span>
                      )}
                    </div>
                  </div>
                  <span className="meta-date">{formatDate(def.created_at)}</span>
                </div>
              </li>
            ))}
          </ul>
        )}
      </section>
    </div>
  )
}
```
